131  std::set<int> visited;

133  // Count changes
134  int nFlips = 0;
135 
136  while (visited.size() != normals.size()) {
137  // Traverse a connected component
138  if (queue.empty()) {
139  if (!visited.size()) {
140  // Initialize queue with one random point
141  queue.push(rand() % normals.size()); // TODO: pick point with low curvature
142  } else {
143  // Expand queue with first unvisited point
144  for (int i = 0; i != normals.size() && queue.empty(); ++i) {
145  // if unvisited, use
146  if (visited.find(i) == visited.end())
147  queue.push(i); // enqueue
148  } //...for each point
149  } //...next component
150 
151  // Set visited
152  visited.insert(queue.front());
153  } //...if queue empty
154 
155  // While points to visit exist
156  while (!queue.empty()) {
157  // Read next point from queue
158  int const pointId = queue.front();
159  // Remove point from queue
160  queue.pop();
161 
162  // Fetch neighbours
163  NeighboursT::const_iterator const iter = neighbours.find(pointId);
164  // Check, if any neighbours
165  if (iter == neighbours.end()) {
166  //std::cerr << "Could not find neighbours of point " << pointId << "\n";
167  continue;
168  }
169 
170  for (int const neighbourId : iter->second) {
171  // If unvisited
172  if (visited.find(neighbourId) == visited.end()) {
173  // Enqueue for next level
174  queue.push(neighbourId);
175  // Mark visited
176  visited.insert(neighbourId);
177 
178  // Flip neighbour normal, if not same direction as precursor point
179  if (normals.row(pointId).dot(normals.row(neighbourId)) < 0.f) {
180  normals.row(neighbourId) *= -1.f;
181  ++nFlips;
182  }
183  } //...if neighbour unvisited
184  } //...for each neighbour of point
185  } //...while points in queue
186  }
